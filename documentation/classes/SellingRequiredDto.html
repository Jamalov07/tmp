	clientId: string

	@ApiProperty({ type: Date })
	@IsNotEmpty()
	@IsDateString()
	date: Date

	@ApiProperty({ type: Boolean })

	sended: boolean

	@IsEnum(SellingStatusEnum)
	status: $Enums.SellingStatusEnum
}
